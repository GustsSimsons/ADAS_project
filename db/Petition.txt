// Create petition table for unapproved tasks
db.serialize(() => {
  db.run(`
    CREATE TABLE IF NOT EXISTS petition (
      petition_id INTEGER PRIMARY KEY AUTOINCREMENT,
      employee_id INTEGER,
      plan_id INTEGER,
      task_description TEXT NOT NULL,
      task_priority TEXT CHECK(task_priority IN ('weak', 'average', 'strong')),
      task_deadline TEXT, -- Format: YYYY-MM-DD
      CS_comments TEXT,
      CTO_comments TEXT,
      FOREIGN KEY (employee_id) REFERENCES employees(employee_id),
      FOREIGN KEY (plan_id) REFERENCES plans(plan_id)
    )
  `, (err) => {
    if (err) {
      console.error('Error creating petition table:', err.message);
    } else {
      console.log('âœ… Petition table created or already exists');
    }
  });
});
